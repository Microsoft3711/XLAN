📡 2. SignalR Test Broadcast (Manual Trigger for Now)
To test live updates, add a button in your dashboard to simulate a change event.

✅ Backend – POST /api/pinghub.cs
csharp


[FunctionName("PingHub")]
public static async Task<IActionResult> Run(
    [HttpTrigger(AuthorizationLevel.Function, "post", Route = "broadcast/refresh")] HttpRequest req,
    [SignalR(HubName = "dashboardHub")] IAsyncCollector<SignalRMessage> signalRMessages)
{
    await signalRMessages.AddAsync(new SignalRMessage {
        Target = "refresh",
        Arguments = new[] { "dashboard update" }
    });

    return new OkResult();
}
